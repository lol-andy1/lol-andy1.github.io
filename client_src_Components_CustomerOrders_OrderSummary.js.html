/** @module OrderSummary */

import React, { useContext } from 'react';
import { OrderContext } from './OrderContext';
import TranslateComponent from '../TopNavBar/Translate';
import { useLanguage } from '../TopNavBar/LanguageContext';

/**
 * Styles for the receipt container.
 * @type {Object}
 */
const receiptContainer = {
  overflowY: 'scroll',
  maxHeight: '17em',
  width: '100%',
  wordWrap: 'break-word',
};

/**
 * Calculates the total price for an item, including base price and toppings.
 *
 * @memberof module:OrderSummary
 * @function
 * @param {Object} item - The item to calculate the total price for.
 * @returns {string} The total price formatted to two decimal places.
 */
function calculateTotalPrice(item) {
    // Calculate the sum of topping prices using reduce
    const toppingsTotalPrice = item.toppings.reduce((total, topping) => total + topping.toppingprice, 0);
    
    // Add the base price and topping prices
    const totalPrice = item.drink.baseprice + toppingsTotalPrice;
  
    return totalPrice.toFixed(2); 
}

/**
 * React component for displaying the order summary.
 *
 * @component
 * @returns {JSX.Element} The OrderSummary component.
 */
const OrderSummary = () => {

  /**
   * The order context to access the current order state.
   * @type {Object}
   */
  const { order } = useContext(OrderContext);

  /**
   * The language context to determine the current language.
   * @type {Object}
   */
  const { language } = useLanguage();

  /**
   * Renders the OrderSummary component.
   *
   * @returns {JSX.Element} JSX representation of the component.
   */
  return (
    <div>
      <div style={receiptContainer}>
        {order && order.map((item, index) => (
          <div key={index}>
            <p>
              <TranslateComponent textToTranslate={`Item ${index + 1}: `} targetLanguage={language} />
              {' '}
              <TranslateComponent textToTranslate={item.drink.drinkname} targetLanguage={language} />
              {' '}
              {item.toppings.length > 0 &&  <TranslateComponent textToTranslate={` with ${item.toppings.map(topping => topping.toppingname).join(', ')}`} targetLanguage={language} />}
              {<TranslateComponent textToTranslate={` and ${item.ice}, ${item.sugar}, ${item.size} Size`} targetLanguage={language} />} -- ${calculateTotalPrice(item)}
            </p>
          </div>
        ))}
      </div>
    </div>
  );
};

export default OrderSummary;
