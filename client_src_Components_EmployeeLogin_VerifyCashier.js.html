/**
 * Verify cashier credentials using OAuth.
 * @param {string} firstName - The first name of the cashier.
 * @param {string} lastName - The last name of the cashier.
 * @returns {Promise<boolean>} A Promise that resolves to true if verification is successful, false otherwise.
 */
export async function VerifyCashierOAuth(firstName, lastName) {
    try {
        const credentials = { first: firstName, last: lastName };
        const response = await fetch(`https://kungfuteaserver09r.onrender.com/verifyCashierOAuth`, {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify(credentials)
        });
        const cashier = await response.json();
        return cashier >= 1;
    } catch (err) {
        console.error(err);
        return false;
    }
}
