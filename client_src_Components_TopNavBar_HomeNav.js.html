/** @module  HomeNav */
import React, { useContext } from "react";
import { Link } from "react-router-dom";
import WeatherButton from "./WeatherButton";
import FontSizeButtons from "../FontSizing/FontSizeButton";
import { FontSizeContext } from "../FontSizing/FontSizeContext";
import MapButton from "./MapButton";
import kftLogo from "../../Images/KFTLogo.png";
import TranslateComponent from "./Translate.js";
import { useLanguage } from "./LanguageContext";

/**
 * Functional component representing the navigation bar for the home page.
 * @returns {JSX.Element} JSX element representing the navigation bar.
 * @memberof module:HomeNav
 */
const HomeNav = ({ }) => {
  const { fontSize } = useContext(FontSizeContext);
  const { language, changeLanguage } = useLanguage();

  /**
   * Handles changing the language when the button is clicked.
   * @function
   * @memberof module:HomeNav
   */
  const handleChangeLanguage = () => {
    changeLanguage();
  };

  return (
    <nav className="navbar navbar-expand-lg navbar-light bg-dark fixed-top">
      <Link className="navbar-brand p-0" to="/">
        <img src={kftLogo} width="50" height="100%" alt="Kung Fu Tea Logo"></img>
      </Link>
      <div className="container-fluid">
        <ul
          className="navbar-nav d-flex flex-row justify-content-center"
          style={{ fontSize: `${fontSize}px` }}
        >
          <li className="nav-item col-6 d-flex justify-content-center me-2">
            <Link
              to="/menu"
              className="btn btn-outline-primary text-center text-white border-2 "
              style={{ width: "100%" }}
            >
              <TranslateComponent textToTranslate = {"Menu"} targetLanguage = {language} />
            </Link>
          </li>
          <li className="nav-item col-6 d-flex justify-content-center">
            <Link
              to="/order"
              className="btn btn-outline-primary text-center text-white border-2" // border-info
              style={{ width: "100%" }}
            >
              <TranslateComponent textToTranslate = {"Order"} targetLanguage = {language} />
            </Link>
          </li>
        </ul>
        <div className = "d-flex flex-grow-1 justify-content-center align-items-center">
          <button onClick={handleChangeLanguage} 
            className="btn btn-outline-primary text-center text-white border-2" 
            style={{ marginLeft: '60%'}}
          >
            {language === 'en' ? 
            <TranslateComponent textToTranslate = {"Spanish"} targetLanguage = {language} /> : 
            <TranslateComponent textToTranslate = {"English"} targetLanguage = {language} />}
          </button>
          
        </div>
        <FontSizeButtons />
        <WeatherButton />
        <MapButton />        
      </div>
    </nav>
  );
};

export default HomeNav;
