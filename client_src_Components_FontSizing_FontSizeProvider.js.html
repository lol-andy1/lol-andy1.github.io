/** @module FontSizeProvider */

import React, { useState } from 'react';
import { FontSizeContext } from './FontSizeContext';

/**
 * FontSizeProvider is a React component that manages and provides the font size state
 * to its descendants using the FontSizeContext.
 *
 * @component
 * @param {Object} props - The properties passed to the FontSizeProvider component.
 * @param {ReactNode} props.children - The child components to be rendered within the provider.
 * @returns {ReactNode} Returns the JSX element with FontSizeContext.Provider wrapping its children.
 */
export function FontSizeProvider({ children }) {
  const [fontSize, setFontSize] = useState(1.2);
  const minFontSize = 1.2;
  const maxFontSize = 2;

  /**
   * Increases the font size by 0.2 units, respecting the maximum font size limit.
   *
   * @function
   * @memberof module:FontSizeProvider
   * @inner
   */
  const increaseFontSize = () => {
    if (fontSize < maxFontSize) {
      setFontSize(fontSize + 0.2);
    }
  };

  /**
   * Decreases the font size by 0.2 units, respecting the minimum font size limit.
   *
   * @function
   * @@memberof module:FontSizeProvider
   * @inner
   */
  const decreaseFontSize = () => {
    if (fontSize > minFontSize) {
      setFontSize(fontSize - 0.2);
    }
  };

  /**
   * Context provider for managing font size state.
   * @typedef {Object} FontSizeContextProvider
   * @property {number} fontSize - The current font size.
   * @property {Function} increaseFontSize - Function to increase the font size.
   * @property {Function} decreaseFontSize - Function to decrease the font size.
   */

  /**
   * @description
   * This module exports a React component called FontSizeProvider, which serves as a
   * provider for managing and providing font size information to its descendants through
   * the FontSizeContext.
   *
   * The FontSizeProvider component takes care of handling font size state, providing
   * context values, and rendering its children within the context provider.
   *
   * @see {@link FontSizeContext}
   */
  return (
    <FontSizeContext.Provider value={{ fontSize, increaseFontSize, decreaseFontSize }}>
      {children}
    </FontSizeContext.Provider>
  );
}
