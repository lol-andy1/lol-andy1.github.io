/** @module  EmployeeRouter */
const express = require("express");
const pool = require("../db");
const router = express.Router();

// checks if the manager's username and password are correct
/**
 * Verifies manager login credentials.
 * @name POST/verifyManagerLogin
 * @function
 * @memberof module:EmployeeRouter
 * @param {Object} req - Express request object containing manager username and password.
 * @param {Object} res - Express response object.
 * @returns {number} - Number of rows that match the credentials.
 */
router.post("/verifyManagerLogin", async(req, res) => {
    try {
        const {user, pass} = req.body;
        const managerQuery = await pool.query("SELECT * FROM employee WHERE username = $1 AND password = $2 AND manager = 't'" , [user, pass]);
        res.json(managerQuery.rowCount);
    } 
    catch (e) {
       console.error(e.message) 
    }
});
