/**

 * @returns {JSX.Element} The JSX element containing the menu board.

const MenuBoard = () => {
  const [drinks, setDrinks] = useState([]);
  const [toppings, setToppings] = useState([]);
  const { fontSize } = useContext(FontSizeContext);

  useEffect(() => {
    const fetchData = async () => {
      try {
        const drinksResponse = await fetch(
          "https://kungfuteaserver09r.onrender.com/getDrinks"
        );
        const drinksData = await drinksResponse.json();
        setDrinks(drinksData);

        const toppingsResponse = await fetch(
          "https://kungfuteaserver09r.onrender.com/getToppings"
        );
        const toppingsData = await toppingsResponse.json();
        setToppings(toppingsData);
      } catch (error) {
        console.error(error);
      }
    };

    fetchData();
  }, []);

  const allDrinks = drinks;

  const column1 = allDrinks.slice(0, Math.ceil((allDrinks.length / 5) * 1));
  const column2 = allDrinks.slice(
    Math.ceil((allDrinks.length / 5) * 1),
    Math.ceil((allDrinks.length / 5) * 2)
  );
  const column3 = allDrinks.slice(
    Math.ceil((allDrinks.length / 5) * 2),
    Math.ceil((allDrinks.length / 5) * 3)
  );
  const column4 = allDrinks.slice(
    Math.ceil((allDrinks.length / 5) * 3),
    Math.ceil((allDrinks.length / 5) * 4)
  );
  const column5 = allDrinks.slice(
    Math.ceil((allDrinks.length / 5) * 4),
    Math.ceil((allDrinks.length / 5) * 5)
  );

  };

  const { language } = useLanguage();

  /**
   * Render function for the MenuBoard component.

  return (

          <div style={toppingsContainer}>

                      }}

                    <Card.Img
