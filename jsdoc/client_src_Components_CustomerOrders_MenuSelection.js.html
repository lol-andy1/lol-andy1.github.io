/** @module MenuSelection */

import React, { useState } from 'react';

/**
 * React component for displaying a menu and allowing users to add items to the cart.
 *
 * @component
 * @param {Object} props - The component props.
 * @param {function} props.addToCart - Function to add an item to the cart.
 * @returns {JSX.Element} The MenuSelection component.
 */
const MenuSelection = ({ addToCart }) => {

  /**
   * State to store the menu items.
   * @type {Array<Object>}
   */
  const [menuItems] = useState([
    { id: 1, name: 'Milk Tea', price: 4.5 },
    { id: 2, name: 'Green Tea', price: 4.0 },
    { id: 3, name: 'Boba', price: 0.50 },
    { id: 4, name: 'Lychee Jelly', price: 0.75 },
    { id: 5, name: 'Red Bean', price: 0.75 },
    { id: 6, name: 'Herbal Jelly', price: 0.50 }
    // Add more menu items
  ]);

  /**
   * Handles the action of adding an item to the cart.
   * @memberof module:MenuSelection
   * @param {Object} item - The menu item to be added to the cart.
   * @returns {void}
   */
  const handleAddToCart = (item) => {
    addToCart(item);
  };

  /**
   * Renders the MenuSelection component.
   * @returns {JSX.Element} JSX representation of the component.
   */
  return (
    <div>
      <h2>Menu</h2>
      <ul>
        {menuItems.map((item) => (
          <li key={item.id}>
            {item.name} - ${item.price}
            <button onClick={() => handleAddToCart(item)}>Add to Cart</button>
          </li>
        ))}
      </ul>
    </div>
  );
};

export default MenuSelection;
