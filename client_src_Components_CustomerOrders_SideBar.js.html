/** @module SideBar */

import { Sidebar } from 'flowbite-react';
import { AiFillHome } from 'react-icons/ai';
import { Link } from 'react-router-dom';
import { useContext } from 'react';
import { OrderContext } from './OrderContext';
import { Button } from 'react-bootstrap';
import OrderSummary from './OrderSummary';
import TranslateComponent from '../TopNavBar/Translate';
import { useLanguage } from '../TopNavBar/LanguageContext'; 
import { FontSizeContext } from '../FontSizing/FontSizeContext';

/**
 * Styles for the sidebar container.
 * @type {Object}
 */
const sidebarContainer = {
  overflowY: 'scroll',
  display: 'flex',
  flexDirection: 'row', // Set the direction to column
  height: '100vh',
};

/**
 * Styles for the sidebar with a box shadow.
 * @type {Object}
 */
const sidebarWithShadow = {
  boxShadow: '10px 0 10px rgba(0, 0, 0, 0.1)', // Add a box shadow to the right side of the sidebar
};

/**
 * Styles for the container inside the sidebar.
 * @type {Object}
 */
const container = {
  alignItems: 'left',
  display: 'flex', // Add this to make the icons display in a column
  flexDirection: 'column', // Set the direction to column
  width: '40vh'
};

/**
 * Styles for the home icon.
 * @type {Object}
 */
const homeIconStyle = {
  fontSize: '4rem', // Increase the font size
  // Add space below the icon
  margin: '0 auto',
  height: '30%'
};

/**
 * Styles for the horizontal line in the sidebar.
 * @type {Object}
 */
const horizontalLine = {
  borderBottom: '1px solid #000', // Customize the border properties as needed
  width: '100%', // Set the width to 100% to create a horizontal line
  marginBottom: '1rem',
  marginTop: '1rem',
};

/**
 * Styles for the home text.
 * @type {Object}
 */
const homeText = {
  //fontSize: `${fontSize}rem`,
  marginBottom: '40px', 
  margin: '0 auto',
};

/**
 * React component for the sidebar containing navigation links and order information.
 *
 * @component
 * @returns {JSX.Element} The SideBar component.
 */
function SideBar() {

  /**
   * Context hook to access font size information.
   * @type {Object}
   */
  const { fontSize } = useContext(FontSizeContext);
  
  /**
   * Language hook to access the selected language.
   * @type {Object}
   */
  const { language } = useLanguage();

  /**
   * Object containing the current order and a function to update it.
   *
   * @typedef {Object} OrderContextData
   * @property {Array} order - An array representing the current order.
   * @property {function} setOrder - A function to update the current order.
   */

  /**
   * Retrieve the current order and the function to update it from the OrderContext.
   *
   * @type {OrderContextData}
   */
  const { order, setOrder } = useContext(OrderContext);

  /**
   * Handles the click event for resetting the order.
   * @function
   * @memberof module:SideBar
   */
  const handleReset = () => {
    setOrder([]); 
  };

  /**
   * Renders the JSX for the SideBar component.
   *
   * @returns {JSX.Element} The JSX representation of the SideBar component.
   */
  return (
    <div className= "position-fixed text-center">
      <div style={sidebarContainer}>
        <Sidebar style={sidebarWithShadow}>
          <div style={container}>
            <Link to='/Order' style={homeIconStyle}><AiFillHome  /></Link>
            <p style={{homeText, fontSize: `${fontSize}rem`}}><TranslateComponent textToTranslate = {("Home")} targetLanguage = {language} /></p>
            {/*<Link to='/order' style={homeIconStyle}><IoArrowBackCircle style={homeIconStyle}></IoArrowBackCircle></Link>
            <p style={homeText}>Back</p>*/}
            <Link to={order.length > 0 ? '/Receipt' : '#'} className=" mt-2 btn btn-success" style={{fontSize: `${fontSize}rem`}}><TranslateComponent textToTranslate = {("Checkout")} targetLanguage = {language} /></Link>

            <div style={horizontalLine}></div> {/* Horizontal line */}
            <h1><TranslateComponent textToTranslate = {("Current Order")} targetLanguage = {language} /></h1>
            <p style={{fontSize: `${fontSize}rem`}}><OrderSummary></OrderSummary></p>
            <Button onClick={handleReset} style={{fontSize: `${fontSize}rem`}}><TranslateComponent textToTranslate = {("Reset Order")} targetLanguage = {language} /></Button>
          </div>
        </Sidebar>
      </div>
    </div>
  );
}

export default SideBar;
