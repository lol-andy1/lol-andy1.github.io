/** @module OrderLanding */

import React, { useContext } from "react";
import { Button } from "react-bootstrap";
import SideBar from "./SideBar"; // Update the import statement
import ClassicImage from "../../Images/MenuCategoriesPics/classic.jpg";
import MilkImage from "../../Images/MenuCategoriesPics/milktea.jpg";
import PunchImage from "../../Images/MenuCategoriesPics/punch.jpg";
import YogurtImage from "../../Images/MenuCategoriesPics/yogurt.jpg";
import { Link } from "react-router-dom";
import TranslateComponent from "../TopNavBar/Translate";
import { useLanguage } from "../TopNavBar/LanguageContext";
import { FontSizeContext } from '../FontSizing/FontSizeContext';

/**
 * React component for displaying the landing page for ordering products.
 *
 * @component
 * @returns {JSX.Element} The OrderLanding component.
 */
const OrderLanding = () => {

  /**
   * The font size context to determine the current font size.
   * @type {Object}
   */
  const { fontSize } = useContext(FontSizeContext);

  /**
   * Styles for the buttons.
   * @type {Object}
   */
  const buttonStyles = {
    boxShadow: "none",
    fontSize: `${fontSize}rem`,
    padding: "15px",
    marginTop: "30px",
    margin: "0px",
    width: "200px", // Fixed width for uniform size
  };

  /**
   * Styles for the product images.
   * @type {Object}
   */
  const imageStyles = {
    objectFit: "cover",
    boxShadow: "0 0 20px rgba(0, 0, 0, 0.3)",
    transition: "box-shadow 0.3s ease",
  };

  /**
   * The language context to determine the current language.
   * @type {Object}
   */
  const { language } = useLanguage();

  /**
   * Renders the OrderLanding component.
   * @returns {JSX.Element} JSX representation of the component.
   */
  return (
    <div className="d-flex vh-100">
      <div className="mt-5">
        <SideBar />
      </div>

      <div
        className="text-center w-100"
        style={{ marginTop: "5rem", marginLeft: "20%" }}
      >
        <h1>
          <TranslateComponent
            textToTranslate={"Products"}
            targetLanguage={language}
          />
        </h1>
        <div className="row justify-content-center mt-4 mx-0">
          <div className="col-4">
            <Link to="./Classic">
              <img
                src={ClassicImage}
                style={imageStyles}
                alt="Classic"
                onMouseOver={(e) => {
                  e.target.style.boxShadow = "0 0 30px rgba(0, 0, 0, 0.6)";
                }}
                onMouseOut={(e) => {
                  e.target.style.boxShadow = "0 0 20px rgba(0, 0, 0, 0.3)";
                }}
              />
            </Link>
          </div>
          <div className="col-4">
            <Link to="./Milk">
              <img
                src={MilkImage}
                style={imageStyles}
                alt="Milk"
                onMouseOver={(e) => {
                  e.target.style.boxShadow = "0 0 30px rgba(0, 0, 0, 0.6)";
                }}
                onMouseOut={(e) => {
                  e.target.style.boxShadow = "0 0 20px rgba(0, 0, 0, 0.3)";
                }}
              />
            </Link>
          </div>
        </div>
        <div className="row justify-content-center mt-5 mx-0">
          <div className="col-4">
            <Link to="./Punch">
              <img
                src={PunchImage}
                style={imageStyles}
                alt="Punch"
                onMouseOver={(e) => {
                  e.target.style.boxShadow = "0 0 30px rgba(0, 0, 0, 0.6)";
                }}
                onMouseOut={(e) => {
                  e.target.style.boxShadow = "0 0 20px rgba(0, 0, 0, 0.3)";
                }}
              />
            </Link>
          </div>
          <div className="col-4">
            <Link to="./Yogurt">
              <img
                src={YogurtImage}
                style={imageStyles}
                alt="Yogurt"
                onMouseOver={(e) => {
                  e.target.style.boxShadow = "0 0 30px rgba(0, 0, 0, 0.6)";
                }}
                onMouseOut={(e) => {
                  e.target.style.boxShadow = "0 0 20px rgba(0, 0, 0, 0.3)";
                }}
              />
            </Link>
          </div>
        </div>
        <div className="row mt-3 mx-0">
          <Link to="./Seasonal" >
              <Button variant="primary" style={buttonStyles}>
                <TranslateComponent
                  textToTranslate={"Seasonal"}
                  targetLanguage={language}
                />
              </Button>
          </Link>
        </div>
      </div>
    </div>
  );
};

export default OrderLanding;
