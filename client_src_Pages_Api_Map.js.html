/** @module Map */

import React, { useEffect, useContext } from 'react';
// import './Map.css';
import { FontSizeContext } from '../../Components/FontSizing/FontSizeContext';
import TranslateComponent from '../../Components/TopNavBar/Translate';
import backgroundImage from '../../Images/homeBackground.jpg'
import { useLanguage } from '../../Components/TopNavBar/LanguageContext';

/**
 * Component for displaying a map with a marker at a specific location.
 * @param {Object} props - Component properties.
 * @param {string} props.apiKey - Google Maps API key.
 * @returns {JSX.Element} The rendered map display.
 */
function MapDisplay({ apiKey }) {
  
  useEffect(() => {
    // Create a script element for the Google Maps API
    const script = document.createElement('script');
    script.src = `https://maps.googleapis.com/maps/api/js?key=${apiKey}&libraries=places`;
    script.async = true;
    script.defer = true;
    script.onload = initMap;
    document.head.appendChild(script);

    /**
     * Load Google Maps API and initialize the map.
     * @memberof module:Map
     */
    function initMap() {
      const lat = 30.626188752352444; // Set the latitude here
      const lng = -96.33026466330449; // Set the longitude here

      const mapOptions = {
        center: { lat: lat, lng: lng },
        zoom: 15, // Adjust the zoom level as needed
      };

      const map = new window.google.maps.Map(document.getElementById('map'), mapOptions);

      // Create a marker at the specified location
      const marker = new window.google.maps.Marker({
        map,
        position: { lat: lat, lng: lng },
        title: 'Kung Fu Tea',
      });

      // Create an info window
      const infoWindow = new window.google.maps.InfoWindow({
        content: 'KUNG FU TEA COLLEGE STATION', // Replace with your text
      });

      // Open the info window when the marker is clicked
      marker.addListener('click', () => {
        infoWindow.open(map, marker);
      });
    }
  }, [apiKey]);

  const { fontSize } = useContext(FontSizeContext);
  const { language } = useLanguage();

  return (
    <div
      className='d-flex flex-column vh-100 justify-content-center align-items-center'
      style={{
        position: 'relative',
        backgroundImage: `url(${backgroundImage})`,
        backgroundSize: 'cover',
        backgroundPosition: 'center',
        paddingTop: "4.75rem"
      }}
    >
      <div
        style={{
          position: 'absolute',
          top: 0,
          left: 0,
          width: '100%',
          height: '100%',
          background: 'rgba(0, 0, 0, 0.5)', // Adjust the alpha value for darkness
        }}
      ></div>
      <h2 style={{ color:'white',font:'Hamilton',fontSize: `${fontSize * 2}rem`, zIndex: 1 }}>
        <TranslateComponent textToTranslate={'Find us at: 23 Texas Ave, College Station, TX 77840'} targetLanguage={language} />
      </h2>
      <div id='map' style={{ width: '60%', height: '75%', zIndex: 1 }}></div>
    </div>
  );
}

export default MapDisplay;
