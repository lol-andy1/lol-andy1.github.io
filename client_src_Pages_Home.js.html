/** @module Home */

import backgroundImg from "../Images/homeBackground.jpg";
import banner from "../Images/kftbanner.jpg";
import React, { useContext } from 'react';
import { Link } from "react-router-dom";
import TranslateComponent from "../Components/TopNavBar/Translate";
import { useLanguage } from "../Components/TopNavBar/LanguageContext";
import { FontSizeContext } from '../Components/FontSizing/FontSizeContext';

const containerStyle = {
    display: 'flex',
    flexDirection: 'row', // Keep row to align background and content horizontally
    height: '100vh',
    overflow: 'hidden',
}

const backgroundStyle = {
    width: '50%', // Set background width to 50% of the viewport width
    backgroundImage: `url(${backgroundImg})`,
    backgroundPosition: "center top 3rem",
    position: 'relative',
    // filter: 'brightness(1.3) contrast(1.2) saturate(1.2)',
}


const contentContainerStyle = {
    flex: 1, // Take up remaining space
    display: 'flex',
    flexDirection: 'column',
    alignItems: 'center', // Center horizontally
    justifyContent: 'center', // Center vertically
}

const bannerStyle = {
    width: '30rem',
    borderRadius: '1rem',
    margin: '2rem', // Adjust the margin-bottom for spacing
    boxShadow: '0 4px 8px rgba(0, 0, 0, 0.2)',
}

const orderButtonStyle = {
    fontSize: '1rem',
    margin: '1rem',
}

const loginButtonStyle = {
    fontSize: '1rem',
    alignSelf: 'flex-end', // Align to the bottom right
    margin: '1rem', // Add margin to the right and bottom
}

/**
 * Component for the Home page.
 * @returns {JSX.Element} The rendered home page.
 */
const Home = () => {
    const { language } = useLanguage();
    const { fontSize } = useContext(FontSizeContext);

    return (
        <div style={containerStyle}>
            <div style={{ ...backgroundStyle }}>
                
            </div>
            <div style={{ ...contentContainerStyle }}>
                <img src={banner} alt="banner" style={{ ...bannerStyle }} />
                <h3 style={{ ...orderButtonStyle, fontSize: `${fontSize}rem` }} >
                    {<TranslateComponent textToTranslate={'Welcome to Kung Fu Tea College Station!'} targetLanguage={language} />}
                </h3>
                <Link to="/Order" className="btn btn-dark btn-sm rounded-pill" style={{ ...orderButtonStyle, fontSize: `${fontSize}rem` }}>
                    <TranslateComponent textToTranslate={'Order Now'} targetLanguage={language} />
                </Link>
                
            </div>
            <Link to="/employeeLogin" className="btn btn-primary btn-sm" style={{ ...loginButtonStyle,fontSize: `${fontSize}rem` }}>
                    {<TranslateComponent textToTranslate={'Employee Login'} targetLanguage={language} />}
            </Link>
        </div>
    );
}

export default Home;
